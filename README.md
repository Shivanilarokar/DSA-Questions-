def fibonacci(n):
    """Returns a list of Fibonacci numbers up to n"""
    fib_sequence = [0, 1]
    while fib_sequence[-1] < n:
        fib_sequence.append(fib_sequence[-1] + fib_sequence[-2])
    return fib_sequence[:-1]

print(fibonacci(10))  # Output: [0, 1, 1, 2, 3, 5, 8]
