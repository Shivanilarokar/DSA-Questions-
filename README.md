def two_sum(nums, target):
    # Example input
    nums = [2, 7, 11, 15]
    target = 9
    for i in range(len(nums)):
        for j in range(i + 1, len(nums)):
            if nums[i] + nums[j] == target:
                return [i, j]

print(two_sum(nums, target))  # Output: [0, 1]  # Because nums[0] + nums[1] == 9
