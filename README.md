# Example usage of a sorting algorithm
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]

sample_array = [64, 34, 25, 12, 22, 11, 90]
bubble_sort(sample_array)
print("Sorted array is:", sample_array)
