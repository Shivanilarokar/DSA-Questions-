# Example Question: Find the maximum element in an array
def find_max(arr):
    max_element = arr[0]
    for num in arr:
        if num > max_element:
            max_element = num
    return max_element

# Sample usage
array = [1, 3, 5, 7, 9]
max_value = find_max(array)
print(f"The maximum value in the array is: {max_value}")
